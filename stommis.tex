\begin{frame}
  \frametitle{Проблема и мотивация}
  \begin{itemize}
    \item В стоматологии результат приёма фиксируется в структурированной электронной карте: диагнозы, процедуры, материалы, зубная формула, рекомендации.
    \item Во время лечения врачу неудобно работать с компьютером: руки заняты инструментами, используются перчатки, внимание сосредоточено на пациенте.
    \item Поэтому значительная часть заполнения карты откладывается на конец приёма, что увеличивает нагрузку и риск потери деталей.
    \item Цель проекта: обеспечить голосовое управление картой \textbf{в процессе лечения} без постоянного взаимодействия с интерфейсом.
  \end{itemize}

\vspace{6pt}
\textbf{Проблема:} нужно обрабатывать только команды врача, не распознавая весь аудиопоток постоянно.

\pdfcomment{Основная проблема не в том, что врач вообще не может заполнить карту. Обычно он может сделать это после лечения. Проблема в том, что во время лечения взаимодействовать с компьютером неудобно: руки заняты, есть перчатки и инструменты. Поэтому мы хотим дать возможность вносить данные в карту голосом прямо в процессе лечения.}
\end{frame}

\begin{frame}
  \frametitle{Постановка задачи}
  \textbf{Цель:} разработать прототип подсистемы голосового управления для МИС «СТОММИС», позволяющий врачу работать с медицинской картой голосом в процессе лечения.

  \vspace{6pt}

  \textbf{Задачи:}
  \begin{enumerate}
    \item Провести обзор существующих решений.
    \item Спроектировать архитектуру подсистемы голосового управления.
    \item Реализовать клиентский модуль захвата аудио и VAD — выделения участков речи и тишины в браузере.
    \item Интегрировать серверный контур: ASR — распознавание речи, классификацию команд, LLM-интерпретацию и REST API МИС.
    \item Провести экспериментальную оценку и апробацию прототипа.

    \pdfcomment{Здесь VAD отвечает за то, чтобы не отправлять весь поток, ASR переводит аудио в текст, бинарный классификатор решает “команда или речевой мусор”, а LLM уже преобразует команду в структурированное действие для МИС.}
    \pdfcommenet{Работа строится как полный инженерный цикл: сначала обзор, затем архитектура, затем реализация клиента и серверного контура, и в конце экспериментальная оценка.}
  \end{enumerate}
\end{frame}


\begin{frame}
  \frametitle{Эволюция подхода}
  \begin{tabular}{|p{0.26\linewidth}|p{0.32\linewidth}|p{0.32\linewidth}|}
    \hline
    \textbf{Подход} & \textbf{Преимущество} & \textbf{Ограничение} \\
    \hline
    \textit{Push-to-talk} & Простая реализация, понятный момент записи & Неудобно во время лечения: руки заняты \\
    \hline
    \textit{Wake word} & Голосовая активация без кнопки & Нужно часто повторять слово активации; ошибки в шуме \\
    \hline
    \textit{Command/not-command} classifier & Автоматически отличает команды от речевого мусора & Требует обученной модели и оценки качества \\
    \hline
    \end{tabular}

    \vspace{8pt}

    \textbf{Итоговый выбор:} серверная классификация транскрипции после ASR перед вызовом LLM.
    \pdfcomment{Мы начали с кнопки, потому что это просто. Потом пробовали wake word, но постоянное повторение слова активации неудобно для врача. Поэтому финально перешли к классификации распознанной фразы: команда это или речевой мусор.}
    \pdfcomment{Мы не пришли сразу к финальному решению. Сначала был push-to-talk, потом wake word. Но оба варианта неудобны в реальном приёме. Поэтому итоговый подход — не заставлять врача произносить слово активации, а автоматически классифицировать распознанную фразу как команду или речевой мусор.}
\end{frame}

\begin{frame}
  \frametitle{Обзор и отличие от типовых решений}

  \begin{itemize}
    \item Типовые решения часто сводятся к диктовке: речь переводится в текст и сохраняется как протокол.
    \item Голосовые ассистенты общего назначения не привязаны к структуре стоматологической карты и справочникам МИС.
    \item Wake word / KWS позволяет запускать голосовой режим без кнопки, но требует повторения слова активации и нестабилен в шумной среде.
  \end{itemize}

  \vspace{6pt}
  \textbf{Отличие предлагаемого подхода:}
  \begin{itemize}
    \item результатом является не текст, а \textbf{структурированное действие} в МИС;
    \item после ASR используется бинарный классификатор \textit{command/not-command};
    \item LLM вызывается только для фраз, классифицированных как команды;
    \item результат LLM ограничивается JSON-контрактом и проверяется валидатором.
  \end{itemize}

  \pdfcomment{Мы не решаем задачу “записать всё, что сказал врач”. Наша задача — понять, является ли фраза командой для системы, и если да, преобразовать её в действие внутри МИС «СТОММИС».}
\end{frame}

\begin{frame}
  \frametitle{Архитектура решения}

  \begin{center}
    \small
    \textbf{Browser}
    $\rightarrow$ VAD
    $\rightarrow$ audio segment
    $\rightarrow$ Flask
    $\rightarrow$ ASR
    $\rightarrow$ classifier
    $\rightarrow$ LLM
    $\rightarrow$ validation
    $\rightarrow$ REST API МИС
  \end{center}

  \vspace{8pt}

  \begin{itemize}
    \item \textbf{Клиент:} захват микрофона, выделение речевых сегментов через VAD, отправка сегмента на сервер.
    \item \textbf{Сервер:} транскодирование аудио, ASR, классификация транскрипции \textit{command/not-command}.
    \item \textbf{ИИ-модуль:} LLM интерпретирует только команды и возвращает структурированное действие.
    \item \textbf{Безопасность:} JSON-контракт, whitelist интентов/слотов, валидация перед изменением карты.
  \end{itemize}

  \vspace{6pt}
  \textbf{Ключевая идея:} дорогостоящая интерпретация запускается только для команд.

\end{frame}

% Implementation details

\begin{frame}
  \frametitle{Серверный конвейер обработки}
  \begin{enumerate}
    \item Приём аудиосегмента из браузера: \texttt{multipart/form-data}, поле \texttt{file}.
    \item Нормализация формата: FFmpeg $\rightarrow$ Ogg/Opus.
    \item Распознавание речи: Yandex SpeechKit $\rightarrow$ текст.
    \item Фильтрация транскрипции: бинарный классификатор \textit{command/not-command}.
    \item Если фраза является командой: LLM формирует структурированное действие.
    \item Валидатор проверяет JSON, интенты и параметры.
    \item Действие применяется через REST API МИС «СТОММИС».
  \end{enumerate}

  \vspace{6pt}
  \textbf{Если классификатор возвращает \textit{not-command}, обработка останавливается до вызова LLM.}

  \pdfcomment{Здесь важно, что ASR ещё вызывается для речевого сегмента, но LLM уже не вызывается для всего подряд. Это снижает количество дорогой семантической обработки и уменьшает риск ложных действий.}
\end{frame}

\begin{frame}
  \frametitle{Идентификация команд по транскрипции}

  \begin{itemize}
    \item После ASR система получает текст речевого сегмента.
    \item Бинарный классификатор решает задачу:
    \[
      text \rightarrow \{\textit{command}, \textit{not-command}\}
    \]
    \item Используемые признаки:
    \begin{itemize}
      \item TF-IDF по символьным n-граммам;
      \item доменные признаки: номер зуба, слова «зуб», «челюсть», диапазоны «со X по Y».
    \end{itemize}
    \item Модель: логистическая регрессия с балансировкой классов.
  \end{itemize}

  \vspace{6pt}
  \textbf{Цель:} не вызывать LLM для речевого мусора.

  \pdfcomment{Мы специально выбрали бинарную классификацию, потому что задача “команда или не команда” проще, чем сразу полная интерпретация. Это лёгкий и быстрый фильтр перед LLM.}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Устойчивость к ошибкам ASR/LLM}

  \begin{itemize}
    \item Первый фильтр: классификатор \textit{command/not-command} отбрасывает речевой мусор до вызова LLM.
    \item Ограничение допустимых действий: whitelist интентов и допустимых параметров.
    \item Жёсткий формат ответа LLM: только JSON по контракту \enquote{intent + slots}.
    \item Валидация перед выполнением: JSON $\rightarrow$ схема $\rightarrow$ проверка слотов по справочникам.
    \item Fail-safe: при низкой уверенности или ошибке валидации действие не применяется.
  \end{itemize}

  \vspace{4pt}
  \begin{center}
    \small
    ASR error $\rightarrow$ classifier / validator $\rightarrow$ no unsafe action
  \end{center}
\end{frame}

\begin{frame}
  \frametitle{Экспериментальное исследование}
  \begin{itemize}
    \item Площадка: тестовая среда и предварительная апробация в СПб ГБУЗ «СП №8».
    \item Сравнение двух режимов:
    \begin{enumerate}
      \item \textbf{Baseline:} непрерывная обработка аудиопотока/чанков.
      \item \textbf{Предложенный режим:} VAD в браузере + ASR + классификатор \textit{command/not-command}.
    \end{enumerate}
    \item Метрики:
    \begin{itemize}
      \item объём аудио, отправленного/обработанного системой;
      \item задержка \enquote{конец команды $\rightarrow$ действие в МИС};
      \item precision/recall/$F_1$ классификатора;
      \item число обращений к LLM после фильтрации.
    \end{itemize}
  \end{itemize}
  \pdfcomment{Мы сравниваем не только время, но и то, насколько уменьшается объём обработки и насколько хорошо classifier отделяет команды от некоманд.}
\end{frame}

\begin{frame}
  \frametitle{Результаты: снижение объёма ASR}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item За 10 минут приёма baseline обрабатывает около $600$ секунд аудио.
        \item В предложенном режиме обрабатывается около $18$--$30$ секунд командных сегментов.
        \item Относительная стоимость ASR оценивается как:
        \[
          \frac{18\text{--}30}{600} \approx 0.03\text{--}0.05
        \]
        \item Типовая задержка команды: $\approx 2$--$3$ секунды.
      \end{itemize}
    \end{column}

    \begin{column}{0.45\textwidth}
      \centering
      \includegraphics[width=\linewidth]{audio_volume.pdf}
      \vspace{2pt}

      {\scriptsize Секунды аудио за 10 минут приёма.}
    \end{column}
  \end{columns}
  \pdfcomment{Стоимость ASR примерно пропорциональна длительности аудио. Поэтому снижение с 600 секунд до 18–30 секунд даёт около 20–30-кратного снижения стоимости ASR в типовом сценарии.}
\end{frame}

\begin{frame}
  \frametitle{Ограничения и планы}
  \begin{itemize}
    \item Классификатор обучен на синтетически сгенерированном наборе; требуется расширение реальными записями врачей.
    \item Качество зависит от шума, микрофона и ошибок ASR.
    \item Необходимо расширить набор команд и формализовать справочники предметной области.
    \item Планируется стабилизация в полевых условиях и сбор метрик на большем объёме данных.
    \item Возможная оптимизация: локальные модели на сервере клиники и кэширование повторяющихся справочников.
  \end{itemize}
\end{frame}
